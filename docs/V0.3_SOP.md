## Session 接口

```python
# core/session.py

@dataclass
class Session:
    session_id: str
    provider_name: str
    created_at: float
    updated_at: float
    status: str          # active / checkpointed / destroyed
    context: dict        # 跨 Task 上下文（键值对）

class SessionManager:
    def create(provider_name: str, context: dict = None) -> Session
    def get(session_id: str) -> Session | None
    def list(provider_name: str = None) -> list[Session]
    def checkpoint(session_id: str, context: dict = None) -> Session
    def resume(session_id: str) -> Session
    def destroy(session_id: str) -> bool
```

## RuntimeRegistry 接口

```python
# core/runtime_registry.py

class RuntimeRegistry:
    def bind(session_id: str, bridge: Bridge) -> None
    def get_bridge(session_id: str) -> Bridge | None
    def unbind(session_id: str) -> None
    def active_sessions() -> list[str]
```
